│   │   ├── routes.py        # Routes spécifiques à Plex Editor (celles de votre routes.py actuel)
│   │   ├── utils.py         # Fonctions utilitaires spécifiques à Plex Editor (votre utils.py actuel)
│   │   └── templates/       # Sous-dossier pour les templates de Plex Editor
│   │       └── plex_editor/ # (pour éviter les conflits de noms, ex: plex_editor/index.html)

│   │           └── select_user.html

│       ├── routes.py        # Routes pour Seedbox UI (votre app.py de SeedboxWebUI adapté)
│       ├── utils.py         # Fonctions utilitaires pour Seedbox UI (si différentes)
│       └── templates/
│           └── seedbox_ui/  # (pour éviter les conflits, ex: seedbox_ui/index.html)
│               └── index.html

└── run.py                   # Point d'entrée pour lancer l'application Flask globale
